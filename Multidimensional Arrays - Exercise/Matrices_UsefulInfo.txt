•	Създаване на празна матрица
	Цели числа: int[,] matrix = new int[rows,cols];
	Дробни числа: double[,] matrix = new double[rows,cols];
	Текстове: String[,] matrix = new String[rows,cols];
	Символи: char[,] matrix = new char[rows,cols];

•	Движения в матрица
	Долу: ред + 1
	Горе: ред – 1
	Ляво: колона – 1
	Дясно: колона + 1


•	Валидиране на позиция в матрица: ред >= 0 && ред < брой редове && колона >= 0 && колона < брой колони
